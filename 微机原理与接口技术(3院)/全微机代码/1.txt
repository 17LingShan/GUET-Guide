DATA SEGMENT
STRING1 DB 'HAPPY ASM'
LEN EQU $-STRING1
STRING2 DB 'HAPPY ASM' 
MSG1 DB 'MATCH!',0DH,0AH,'$'
MSG2 DB 'NO MATCH!',0DH,0AH,'$'
DATA ENDS

CODE SEGMENT
    ASSUME CS:CODE,DS:DATA,ES:DATA
START:
    MOV AX,DATA
    MOV DS,AX
    MOV ES,AX
    LEA SI,STRING1
    LEA DI,STRING2
    MOV CX,LEN
COMPARE:MOV AL,[SI]
		MOV BL,[DI]
		CMP AL,BL
		JNE DISMATCH
		INC SI
		INC DI
LOOP COMPARE
MATCH:LEA DX,MSG1
	  JMP DISP
DISMATCH:LEA DX,MSG2
DISP:    MOV AH,09H
		INT 21H 
    MOV AH,4CH
    INT 21H
CODES ENDS
    END START
